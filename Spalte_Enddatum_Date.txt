VAR DatumText =
    SUBSTITUTE(
        SUBSTITUTE(
            SUBSTITUTE(
                SUBSTITUTE(
                    SUBSTITUTE(
                        SUBSTITUTE(
                            SUBSTITUTE(
                                SUBSTITUTE(
                                    'Roadmap_ITIL'[Enddatum],
                                    "Montag, ", ""
                                ),
                                "Dienstag, ", ""
                            ),
                            "Mittwoch, ", ""
                        ),
                        "Donnerstag, ", ""
                    ),
                    "Freitag, ", ""
                ),
                "Samstag, ", ""
            ),
            "Sonntag, ", ""
        ),
        ".", "-"
    )
RETURN
    IFERROR(DATEVALUE(TRIM(DatumText)), BLANK())
